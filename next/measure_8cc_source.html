108 void MeasureHandler::render(ViewInterface::Ptr const& vi,
109  cairo_t* cr,
110  Scroom::Utils::Rectangle<double> presentationArea,
111  int zoom)
112 {
113  UNUSED(vi);
114 
115  if(selection)
116  {
117  auto aspectRatio = vi->getCurrentPresentation()->getAspectRatio();
118  auto start = Scroom::Utils::Point<int>(selection->start) - presentationArea.getTopLeft();
119  auto end = Scroom::Utils::Point<int>(selection->end) - presentationArea.getTopLeft();
120 
121  if(zoom >= 0)
122  {
123  const int pixelSize = 1 << zoom;
124  start *= pixelSize;
125  start *= aspectRatio;
126  end *= pixelSize;
127  end *= aspectRatio;
128  }
129  else
130  {
131  const int pixelSize = 1 << -zoom;
132  start *= aspectRatio;
133  start /= pixelSize;
134  end *= aspectRatio;
135  end /= pixelSize;
136  }
137 
138  cairo_set_line_width(cr, 1);
139  cairo_set_source_rgb(cr, 0.75, 0, 0); // Dark Red
140  drawCross(cr, start);
141  drawCross(cr, end);
142  cairo_stroke(cr);
143  cairo_set_source_rgb(cr, 1, 0, 0); // Red
144  cairo_move_to(cr, start.x, start.y);
145  cairo_line_to(cr, end.x, end.y);
146  cairo_stroke(cr);
147  }
148 }
