172 void DataFetcher::operator()()
173 {
174  // printf("Attempting to fetch bitmap data for tileRow %d...\n", currentRow);
175  QueueJumper::Ptr qj = QueueJumper::create();
176 
177  threadPool->schedule(qj, REDUCE_PRIO, queue);
178 
179  CompressedTileLine& tileLine = layer->getTileLine(currentRow);
180  std::vector<Tile::Ptr> tiles;
181  for(int x = 0; x < horTileCount; x++)
182  {
183  CompressedTile::Ptr ti = tileLine[x];
184  Scroom::Utils::Stuff s = ti->initialize();
185  tiles.push_back(ti->getTileSync());
186  }
187  int lineCount = std::min(TILESIZE, height - currentRow * TILESIZE);
188 
189  sp->fillTiles(currentRow * TILESIZE, lineCount, TILESIZE, 0, tiles);
190 
191  for(int x = 0; x < horTileCount; x++)
192  {
193  tileLine[x]->reportFinished();
194  }
195 
196  currentRow++;
197  if(currentRow < verTileCount)
198  {
199  DataFetcher successor(*this);
200  if(!qj->setWork(successor))
201  {
202  threadPool->schedule(successor, DATAFETCH_PRIO, queue);
203  }
204  }
205  else
206  {
207  sp->done();
208  }
209 }
