41 bool TiffPresentation::load(const std::string& fileName_)
42 {
43  try
44  {
45  this->fileName = fileName_;
46  tif = TIFFOpen(fileName_.c_str(), "r");
47  if(!tif)
48  {
49  // Todo: report error
50  printf("PANIC: Failed to open file %s\n", fileName_.c_str());
51  return false;
52  }
53 
54  uint16 spp_ = 0;
55  if(1 != TIFFGetField(tif, TIFFTAG_SAMPLESPERPIXEL, &spp_))
56  {
57  spp_ = 1; // Default value, according to tiff spec
58  }
59  if(spp_ != 1 && spp_ != 3 && spp_ != 4)
60  {
61  printf("PANIC: Samples per pixel is neither 1 nor 3 nor 4, but %d. Giving up\n", spp_);
62  return false;
63  }
64  this->spp = spp_;
65 
66  TIFFGetFieldChecked(tif, TIFFTAG_IMAGEWIDTH, &width);
67  TIFFGetFieldChecked(tif, TIFFTAG_IMAGELENGTH, &height);
68 
69  uint16 bps_ = 0;
70  if(1 != TIFFGetField(tif, TIFFTAG_BITSPERSAMPLE, &bps_))
71  {
72  if(spp == 1)
73  {
74  bps_ = 1;
75  }
76  else
77  {
78  bps_ = 8;
79  }
80  }
81  else
82  {
83 
84  if(spp == 3)
85  {
86  if(bps_ != 8)
87  {
88  printf("PANIC: Bits per sample is not 8, but %d. Giving up\n", bps_);
89  return false;
90  }
91  }
92  }
93  this->bps = bps_;
94 
95  Colormap::Ptr originalColormap;
96 
97  uint16* r;
98 
99  uint16* g;
100 
101  uint16* b;
102  int result = TIFFGetField(tif, TIFFTAG_COLORMAP, &r, &g, &b);
103  if(result == 1)
104  {
105  originalColormap = Colormap::create();
106  originalColormap->name = "Original";
107  size_t count = 1UL << bps;
108  originalColormap->colors.resize(count);
109 
110  for(size_t i = 0; i < count; i++)
111  {
112  originalColormap->colors[i] = Color(1.0 * r[i] / 0xFFFF, 1.0 * g[i] / 0xFFFF, 1.0 * b[i] / 0xFFFF);
113  }
114 
115  colormapHelper = ColormapHelper::create(originalColormap);
116  }
117 
118  uint16 photometric;
119  TIFFGetFieldChecked(tif, TIFFTAG_PHOTOMETRIC, &photometric);
120  switch(photometric)
121  {
122  case PHOTOMETRIC_MINISBLACK:
123  if(originalColormap)
124  {
125  printf("WEIRD: Tiff contains a colormap, but photometric isn't palette\n");
126  }
127 
128  if(bps == 1 || bps == 8)
129  {
130  colormapHelper = MonochromeColormapHelper::create(2);
131  }
132  else
133  {
134  colormapHelper = MonochromeColormapHelper::create(1 << bps);
135  }
136 
137  properties[MONOCHROME_COLORMAPPABLE_PROPERTY_NAME] = "";
138  break;
139 
140  case PHOTOMETRIC_MINISWHITE:
141  if(originalColormap)
142  {
143  printf("WEIRD: Tiff contains a colormap, but photometric isn't palette\n");
144  }
145 
146  if(bps == 1 || bps == 8)
147  {
148  colormapHelper = MonochromeColormapHelper::createInverted(2);
149  }
150  else
151  {
152  colormapHelper = MonochromeColormapHelper::createInverted(1 << bps);
153  }
154 
155  properties[MONOCHROME_COLORMAPPABLE_PROPERTY_NAME] = "";
156  break;
157 
158  case PHOTOMETRIC_PALETTE:
159  if(!originalColormap)
160  {
161  printf("WEIRD: Photometric is palette, but tiff doesn't contain a colormap\n");
162  colormapHelper = ColormapHelper::create(1 << bps);
163  }
164  break;
165 
166  case PHOTOMETRIC_RGB:
167  if(originalColormap)
168  {
169  printf("WEIRD: Tiff contains a colormap, but photometric isn't palette\n");
170  }
171  break;
172 
173  case PHOTOMETRIC_SEPARATED:
174  if(originalColormap)
175  {
176  printf("WEIRD: Tiff contains a colormap, but photometric isn't palette\n");
177  }
178  break;
179 
180  default:
181  printf("PANIC: Unrecognized value for photometric\n");
182  return false;
183  }
184 
185  float resolutionX;
186  float resolutionY;
187  uint16 resolutionUnit;
188 
189  if(TIFFGetField(tif, TIFFTAG_XRESOLUTION, &resolutionX) && TIFFGetField(tif, TIFFTAG_YRESOLUTION, &resolutionY)
190  && TIFFGetField(tif, TIFFTAG_RESOLUTIONUNIT, &resolutionUnit))
191  {
192  if(resolutionUnit != RESUNIT_NONE)
193  {
194  // Fix aspect ratio only
195  float base = std::max(resolutionX, resolutionY);
196  resolutionX = resolutionX / base;
197  resolutionY = resolutionY / base;
198  }
199 
200  transformationData = TransformationData::create();
201  transformationData->setAspectRatio(1 / resolutionX, 1 / resolutionY);
202  }
203  else
204  {
205  resolutionX = 1;
206  resolutionY = 1;
207  }
208  printf("This bitmap has size %d*%d, aspect ratio %.1f*%.1f\n", width, height, 1 / resolutionX, 1 / resolutionY);
209 
210  LayerSpec ls;
211  if(spp == 4 && bps == 8)
212  {
213  auto cmykOperations = OperationsCMYK32::create();
214  pipetteLayerOperation = cmykOperations;
215  ls.push_back(cmykOperations);
216  properties[PIPETTE_PROPERTY_NAME] = "";
217  }
218  else if(spp == 4 && bps == 4)
219  {
220  auto cmykOperations = OperationsCMYK16::create();
221  pipetteLayerOperation = cmykOperations;
222  ls.push_back(cmykOperations);
223  ls.push_back(OperationsCMYK32::create());
224  properties[PIPETTE_PROPERTY_NAME] = "";
225  }
226  else if(spp == 4 && bps == 2)
227  {
228  auto cmykOperations = OperationsCMYK8::create();
229  pipetteLayerOperation = cmykOperations;
230  ls.push_back(cmykOperations);
231  ls.push_back(OperationsCMYK32::create());
232  properties[PIPETTE_PROPERTY_NAME] = "";
233  }
234  else if(spp == 4 && bps == 1)
235  {
236  auto cmykOperations = OperationsCMYK4::create();
237  pipetteLayerOperation = cmykOperations;
238  ls.push_back(cmykOperations);
239  ls.push_back(OperationsCMYK32::create());
240  properties[PIPETTE_PROPERTY_NAME] = "";
241  }
242  else if(spp == 3 && bps == 8)
243  {
244  auto rgbOperations = Operations24bpp::create();
245  pipetteLayerOperation = rgbOperations;
246  ls.push_back(rgbOperations);
247  properties[PIPETTE_PROPERTY_NAME] = "";
248  }
249  else if(bps == 2 || bps == 4 || photometric == PHOTOMETRIC_PALETTE)
250  {
251  ls.push_back(Operations::create(colormapHelper, bps));
252  ls.push_back(OperationsColormapped::create(colormapHelper, bps));
253  properties[COLORMAPPABLE_PROPERTY_NAME] = "";
254  }
255  else if(bps == 1)
256  {
257  ls.push_back(Operations1bpp::create(colormapHelper));
258  ls.push_back(Operations8bpp::create(colormapHelper));
259  }
260  else if(bps == 8)
261  {
262  ls.push_back(Operations8bpp::create(colormapHelper));
263  }
264  else
265  {
266  printf("PANIC: %d bits per pixel not supported\n", bps);
267  return false;
268  }
269 
270  tbi = createTiledBitmap(width, height, ls);
271  tbi->setSource(shared_from_this<SourcePresentation>());
272  return true;
273  }
274  catch(const std::exception& ex)
275  {
276  printf("PANIC: %s\n", ex.what());
277  return false;
278  }
279 }

367 void TiffPresentation::fillTiles(int startLine, int lineCount, int tileWidth, int firstTile, std::vector<Tile::Ptr>& tiles)
368 {
369  // printf("Filling lines %d to %d, tile %d to %d (tileWidth = %d)\n",
370  // startLine, startLine+lineCount,
371  // firstTile, (int)(firstTile+tiles.size()),
372  // tileWidth);
373 
374  const auto startLine_ = static_cast<uint32>(startLine);
375  const auto firstTile_ = static_cast<size_t>(firstTile);
376  const auto scanLineSize = static_cast<size_t>(TIFFScanlineSize(tif));
377  const auto tileStride = static_cast<size_t>(tileWidth * spp * bps / 8);
378  std::vector<byte> row(scanLineSize);
379 
380  const size_t tileCount = tiles.size();
381  auto dataPtr = std::vector<byte*>(tileCount);
382  for(size_t tile = 0; tile < tileCount; tile++)
383  {
384  dataPtr[tile] = tiles[tile]->data.get();
385  }
386 
387  for(size_t i = 0; i < static_cast<size_t>(lineCount); i++)
388  {
389  TIFFReadScanline(tif, row.data(), static_cast<uint32>(i) + startLine_);
390 
391  for(size_t tile = 0; tile < tileCount - 1; tile++)
392  {
393  memcpy(dataPtr[tile], row.data() + (firstTile_ + tile) * tileStride, tileStride);
394  dataPtr[tile] += tileStride;
395  }
396  memcpy(dataPtr[tileCount - 1],
397  row.data() + (firstTile_ + tileCount - 1) * tileStride,
398  scanLineSize - (firstTile_ + tileCount - 1) * tileStride);
399  dataPtr[tileCount - 1] += tileStride;
400  }
401 }

443 PipetteLayerOperations::PipetteColor TiffPresentation::getPixelAverages(Scroom::Utils::Rectangle<int> area)
444 {
445  require(pipetteLayerOperation);
446 
447  Scroom::Utils::Rectangle<int> presentationArea = getRect().toIntRectangle();
448  area = area.intersection(presentationArea);
449 
450  Layer::Ptr bottomLayer = tbi->getBottomLayer();
451  PipetteLayerOperations::PipetteColor pipetteColors;
452 
453  int totalPixels = area.getWidth() * area.getHeight();
454 
455  if(totalPixels == 0)
456  {
457  return {};
458  }
459 
460  // Get start tile (tile_pos_x_start, tile_pos_y_start)
461  int tile_pos_x_start = area.getLeft() / TILESIZE;
462  int tile_pos_y_start = area.getTop() / TILESIZE;
463 
464  // Get end tile (tile_pos_x_end, tile_pos_y_end)
465  int tile_pos_x_end = (area.getRight() - 1) / TILESIZE;
466  int tile_pos_y_end = (area.getBottom() - 1) / TILESIZE;
467 
468  for(int x = tile_pos_x_start; x <= tile_pos_x_end; x++)
469  {
470  for(int y = tile_pos_y_start; y <= tile_pos_y_end; y++)
471  {
472  ConstTile::Ptr tile = bottomLayer->getTile(x, y)->getConstTileSync();
473  Scroom::Utils::Rectangle<int> tile_rectangle(x * TILESIZE, y * TILESIZE, tile->width, tile->height);
474 
475  Scroom::Utils::Rectangle<int> inter_rect = tile_rectangle.intersection(area);
476  Scroom::Utils::Point<int> base(x * TILESIZE, y * TILESIZE);
477 
478  inter_rect -= base; // rectangle coordinates relative to constTile with topleft corner (0,0)
479 
480  pipetteColors = sumPipetteColors(pipetteColors, pipetteLayerOperation->sumPixelValues(inter_rect, tile));
481  }
482  }
483  return dividePipetteColors(pipetteColors, totalPixels);
484 }
