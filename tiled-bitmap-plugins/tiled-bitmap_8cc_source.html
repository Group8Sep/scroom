147 void TiledBitmap::initialize(const Layer::Ptr bottom)
148 {
149  int width = bitmapWidth;
150  int height = bitmapHeight;
151  unsigned int i = 0;
152  LayerOperations::Ptr lo = ls[i];
153  Scroom::MemoryBlobs::PageProvider::Ptr provider = bottom->getPageProvider();
154 
155  registrations.push_back(bottom->registerObserver(shared_from_this<TileInitialisationObserver>()));
156  layers.push_back(bottom);
157 
158  Layer::Ptr prevLayer = bottom;
159  LayerOperations::Ptr prevLo = lo;
160 
161  width = (width + 7) / 8; // Round up
162  height = (height + 7) / 8;
163  i++;
164 
165  while(std::max(width, height) > TILESIZE / 4)
166  {
167  if(i < ls.size())
168  {
169  lo = ls[i];
170  }
171 
172  Layer::Ptr layer = Layer::create(i, width, height, lo->getBpp(), provider);
173  registrations.push_back(layer->registerObserver(shared_from_this<TileInitialisationObserver>()));
174  layers.push_back(layer);
175 
176  connect(layer, prevLayer, prevLo);
177 
178  prevLayer = layer;
179  prevLo = lo;
180  width = (width + 7) / 8; // Round up
181  height = (height + 7) / 8;
182  i++;
183  }
184 }

202 void TiledBitmap::connect(Layer::Ptr const& layer, Layer::Ptr const& prevLayer, LayerOperations::Ptr prevLo)
203 {
204  int horTileCount = prevLayer->getHorTileCount();
205  int verTileCount = prevLayer->getVerTileCount();
206 
207  std::vector<LayerCoordinator::Ptr> coordinators_;
208 
209  for(int j = 0; j < verTileCount; j++)
210  {
211  int voffset = j % 8;
212  if(!voffset)
213  {
214  // New line of target tiles
215  coordinators_.clear();
216  CompressedTileLine& tileLine = layer->getTileLine(j / 8);
217  for(auto& z: tileLine)
218  {
219  LayerCoordinator::Ptr lc = LayerCoordinator::create(z, prevLo);
220  coordinators_.push_back(lc);
221  this->coordinators.push_back(lc);
222  }
223  }
224 
225  for(int i = 0; i < horTileCount; i++)
226  {
227  int hoffset = i % 8;
228  LayerCoordinator::Ptr lc = coordinators_[i / 8];
229  lc->addSourceTile(hoffset, voffset, prevLayer->getTile(i, j));
230  }
231  }
232 }

270 void TiledBitmap::redraw(ViewInterface::Ptr const& vi,
271  cairo_t* cr,
272  Scroom::Utils::Rectangle<double> const& presentationArea,
273  int zoom)
274 {
275  TiledBitmapViewData::Ptr viewData_ = this->viewData[vi];
276  auto scaledRequestedPresentationArea = presentationArea;
277 
278  unsigned int layerNr = 0;
279  while(zoom <= -3 && layerNr < layers.size() - 1)
280  {
281  layerNr++;
282  zoom += 3;
283  scaledRequestedPresentationArea /= 8;
284  }
285  Layer::Ptr layer = layers[layerNr];
286  LayerOperations::Ptr layerOperations = ls[std::min(ls.size() - 1, static_cast<size_t>(layerNr))];
287 
288  const Scroom::Utils::Rectangle<int> actualPresentationArea = layer->getRect();
289  const auto validPresentationArea = scaledRequestedPresentationArea.intersection(actualPresentationArea);
290 
291  const int left = scaledRequestedPresentationArea.getLeft();
292  const int top = scaledRequestedPresentationArea.getTop();
293  const int right = scaledRequestedPresentationArea.getRight();
294  const int bottom = scaledRequestedPresentationArea.getBottom();
295 
296  const int imin = std::max(0, left / TILESIZE);
297  const int imax = (right + TILESIZE - 1) / TILESIZE;
298  const int jmin = std::max(0, top / TILESIZE);
299  const int jmax = (bottom + TILESIZE - 1) / TILESIZE;
300 
301  viewData_->setNeededTiles(layer, imin, imax, jmin, jmax, zoom, layerOperations);
302 
303  const double pixelSize = pixelSizeFromZoom(zoom);
304 
305  layerOperations->initializeCairo(cr);
306 
307  const auto clippedRequestedPresentationArea =
308  scaledRequestedPresentationArea.above(validPresentationArea.getBottom()).leftOf(validPresentationArea.getRight());
309 
310  for(int i = imin; i < imax; i++)
311  {
312  for(int j = jmin; j < jmax; j++)
313  {
314  Scroom::Utils::Point<int> tileIndex(i, j);
315 
316  const auto tileArea = TileAreaForIndex(tileIndex);
317  const auto visibleTileArea = tileArea.intersection(clippedRequestedPresentationArea);
318 
319  const auto tileAreaRect = visibleTileArea - tileArea.getTopLeft();
320  const auto viewAreaRect = (visibleTileArea - clippedRequestedPresentationArea.getTopLeft()) * pixelSize;
321 
322  CompressedTile::Ptr tile = layer->getTile(i, j);
323  TileViewState::Ptr tileViewState = tile->getViewState(vi);
324  Scroom::Utils::Stuff cacheResult = tileViewState->getCacheResult();
325  ConstTile::Ptr t = tile->getConstTileAsync();
326 
327  if(t)
328  {
329  cairo_save(cr);
330  layerOperations->draw(cr, t, tileAreaRect, viewAreaRect, zoom, cacheResult);
331  cairo_restore(cr);
332  }
333  else
334  {
335  layerOperations->drawState(cr, tile->getState(), viewAreaRect);
336  }
337 #ifdef DEBUG_TILES
338  drawTile(cr, tile, viewAreaRect);
339 #endif
340  }
341  }
342 }
